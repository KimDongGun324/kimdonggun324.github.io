for i in range(1, 11):
    s += i   # s = 55

# 홀수만 (1, 3, 5, 7, 9)
for i in range(1, 11, 2):
    print(i)

# 역순 (10, 8, 6, 4, 2, 0)
for i in range(10, -1, -2):
    print(i)
